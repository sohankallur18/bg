Business Case: Efficient Inventory Management

Description: In the fast-paced world of business, effective inventory management is the key to thriving in competitive markets. Imagine a network of warehouses and retail stores connected through various transportation routes. The challenge lies in determining the most cost-effective and efficient way to transport goods between these locations. This involves finding the shortest paths while considering factors like transportation costs, delivery time, and route availability.

Efficient inventory management helps businesses:
Minimize logistics costs.
Avoid overstocking or understocking.
Ensure timely delivery of goods.
Boost customer satisfaction.

By leveraging advanced graph-based algorithms, we can optimize the supply chain and improve decision-making. Each warehouse and retail store is represented as a node, while transportation routes between them are modeled as edges with weights denoting transportation costs or times.

#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <algorithms>
const int INF = 1e9;
void printDistances(vector& dist, int n) {
for (int i = 0; i < n; i++) {
if (dist[i] == INF) cout << "INF ";
else cout << dist[i] << " ";
}
cout << endl;
}
void dijkstra(vector>>& graph, int src, int n) {
vector dist(n, INF);
vector visited(n, false);
dist[src] = 0;
for (int i = 0; i < n; i++) {
int u = -1;
for (int j = 0; j < n; j++) {
if (!visited[j] && (u == -1 || dist[j] < dist[u])) u = j;
}
if (dist[u] == INF) break;
visited[u] = true;
for (auto edge : graph[u]) {
int v = edge.first, weight = edge.second;
if (dist[u] + weight < dist[v]) {
dist[v] = dist[u] + weight;
}
}
}
cout << "Dijkstra's distances from source " << src << ": ";
printDistances(dist, n);
}
void bellmanFord(vector>>& edges, int src, int n) {
vector dist(n, INF);
dist[src] = 0;
for (int i = 1; i < n; i++) {
for (auto edge : edges) {
int u = edge[0], v = edge[1], weight = edge[2];
if (dist[u] != INF && dist[u] + weight < dist[v]) {
dist[v] = dist[u] + weight;
}
}
}
for (auto edge : edges) {
int u = edge[0], v = edge[1], weight = edge[2];
if (dist[u] != INF && dist[u] + weight < dist[v]) {
cout << "Graph contains a negative weight cycle.\n";
return;
}
}
cout << "Bellman-Ford distances from source " << src << ": ";
printDistances(dist, n);
}
void floydWarshall(vector>& graph, int n) {
vector> dist = graph;
for (int k = 0; k < n; k++) {
for (int i = 0; i < n; i++) {
for (int j = 0; j < n; j++) {
if (dist[i][k] != INF && dist[k][j] != INF) {
dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
}
}
}
}
cout << "Floyd-Warshall distance matrix:\n";
for (int i = 0; i < n; i++) {
printDistances(dist[i], n);
}
}
int main() {
int n = 5;
vector>> graph(n);
vector> matrix(n, vector(n, INF));
vector>> edges;
graph[0].push_back({1, 10});
graph[0].push_back({2, 3});
graph[1].push_back({2, 1});
graph[1].push_back({3, 2});
graph[2].push_back({3, 8});
graph[3].push_back({4, 7});
graph[4].push_back({0, 5});
for (int i = 0; i < n; i++) matrix[i][i] = 0;
for (int u = 0; u < n; u++) {
for (auto v : graph[u]) {
matrix[u][v.first] = v.second;
edges.push_back({u, v.first, v.second});
}
}
dijkstra(graph, 0, n);
bellmanFord(edges, 0, n);
floydWarshall(matrix, n);
return 0;
}

Implementation

Overview

Dijkstra's Algorithm:

Designed for finding the shortest path from a single source to all other nodes. It's ideal for networks with non-negative weights.

Bellman-Ford Algorithm:

Extends Dijkstra's capabilities by handling graphs with negative weight edges, enabling better versatility in real-world scenarios.

Floyd-Warshall Algorithm:

Computes shortest paths between all pairs of nodes, making it perfect for analyzing the entire supply chain at once.
